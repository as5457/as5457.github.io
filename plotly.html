get_all_inspections = function(url) {
  
  all_inspections = vector("list", length = 0)
  
  loop_index = 1
  chunk_size = 50000
  DO_NEXT = TRUE
  
  while (DO_NEXT) {
    message("Getting data, page ", loop_index)
    
    all_inspections[[loop_index]] = 
      GET(url,
          query = list(`$order` = "zipcode",
                       `$limit` = chunk_size,
                       `$offset` = as.integer((loop_index - 1) * chunk_size)
                       )
          ) %>%
      content("text") %>%
      fromJSON() %>%
      as_tibble()
    
    DO_NEXT = dim(all_inspections[[loop_index]])[1] == chunk_size
    loop_index = loop_index + 1
  }
  
  all_inspections
  
}

rest_inspec =
  nyc_inspections %>%
  filter(grade %in% c("A", "B", "C"), boro != "Missing") %>% 
  mutate(boro = str_to_title(boro))

What is the frequency of select ethnic restaurants in each borough?

eth = rest_inspec %>% 
  filter (cuisine_description %in% c("Chinese", "French", "Japanese", "Indian", "Italian")) %>% 
  group_by (boro, cuisine_description) %>% 
  summarize (n = n()) %>% 
  ggplot(aes(x = cuisine_description, y=n, fill = boro)) +
  geom_bar(stat="identity") + facet_wrap(~ boro, ncol=2) +
  labs (
    title = "Frequency of Select Ethnic Restaurants By Borough",
        x = "Cuisine Type",
        y = "Frequency"
  ) + theme(axis.text.x = element_text(angle = 75, hjust = 1)) + theme(legend.position="none")

Cleaning the dataset to capture years

rest_clean = rest_inspec %>% 
  separate(grade_date, into = c("year", "month", "date"), sep = "-") %>%
  separate(date, into = c("date", "remove"), sep = 2) %>%
  select (boro, cuisine_description, grade, year, month, date, camis, score) %>%   mutate(year = as.numeric(year))

Let’s find out what the how many restaurants got A’s in each borough by year

rest = rest_clean %>% 
  filter (year > 2013) %>% 
  filter (grade == "A") %>%
  group_by (boro, year) %>% 
  summarize (count = n()) %>% 
  ggplot (aes(x = year, y = count, color = boro)) +
    geom_line () + geom_point() +
  labs (
    title = "Frequency of Total Number of A Grades Assigned",
        x = "Year",
        y = "Frequency"
